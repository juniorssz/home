Vetores de Texto

Visão Geral


Criando um vetor de 1 elemento:

estado <- "Paraná"
estado
## [1] "Paraná"


Contando o número de caracteres do elemento:

nchar(estado)
## [1] 6


Criando um vetor com múltiplos elementos:

estados <- c("Paraná", "Goiania", "Cera", "PA")
estados
## [1] "Paraná"  "Goiania" "Cera"    "PA"


Contando o número de caracteres dos elementos:

nchar(estado)
## [1] 6


Contando o número de elementos do vetor:

length(estados)
## [1] 4


Espaços


Trabalhando com espaços nos extremos:

com_espaco <- " casa "
com_espaco
## [1] " casa "
nchar(com_espaco)
## [1] 6
sem_espaco <- trimws(com_espaco)
sem_espaco
## [1] "casa"
nchar(sem_espaco)
## [1] 4


Trabalhando com espaços internos:

frase <- "R é qualidade de vida"
frase
## [1] "R é qualidade de vida"
nchar(frase)
## [1] 21
frase_sem_espaco <- gsub(pattern = " ", replacement = "", x = frase)
frase_sem_espaco
## [1] "Réqualidadedevida"
nchar(frase_sem_espaco)
## [1] 17


Contando espaços:

nchar("")
## [1] 0
nchar(" ")
## [1] 1
nchar("  ")
## [1] 2
nchar("   ")
## [1] 3


Conversão


Convertendo um vetor numérico para texto:

numeros <- c(1,2,3,4,5)
numeros
## [1] 1 2 3 4 5
class(numeros)
## [1] "numeric"
numeros_fake <- as.character(numeros)
numeros_fake
## [1] "1" "2" "3" "4" "5"
class(numeros_fake)
## [1] "character"


Convertendo um vetor de texto para númerico (formato numérico):

numeros_conv <- as.numeric(numeros_fake)
numeros_conv
## [1] 1 2 3 4 5
class(numeros_conv)
## [1] "numeric"


Convertendo um vetor de texto para numérico (formato texto):

animais <- c("macaco", "cachorro", "girafa")
animais
## [1] "macaco"   "cachorro" "girafa"
conversao <- as.numeric(animais)
## Warning: NAs introduced by coercion
conversao
## [1] NA NA NA


Convertendo um vetor de texto para numérico (formato numérico com separador incorreto):

incorreto <- c("2,6", "4,7", "10,5")
incorreto
## [1] "2,6"  "4,7"  "10,5"
correto <- as.numeric(incorreto)
## Warning: NAs introduced by coercion
correto
## [1] NA NA NA
correto <- gsub(pattern = ",", replacement = ".", incorreto)
correto
## [1] "2.6"  "4.7"  "10.5"
correto <- as.numeric(correto)
correto
## [1]  2.6  4.7 10.5


Extração de texto


Extraindo os 2 primeiros caracteres:

codigos <- c("01-1234", "02-4321", "03-5678", "04-8765")
codigos
## [1] "01-1234" "02-4321" "03-5678" "04-8765"
codigos_2p <- substr(x = codigos, start = 1, stop = 2)
codigos_2p
## [1] "01" "02" "03" "04"


Extraindo os 4 últimos caracteres:

codigos_4u <- substr(x = codigos, start = 4, stop = 8)
codigos_4u
## [1] "1234" "4321" "5678" "8765"


Criando as funções direita() e esquerda() como no Excel:

esquerda <- function(texto, n) {
  out <- substr(x = texto, 
                start = 1, 
                stop = n)
  return(out)
}

codigos_2p <- esquerda(texto = codigos,  n = 2)
codigos_2p
## [1] "01" "02" "03" "04"
direita <- function(texto, n) {
  out <- substr(x = texto, 
                start = nchar(texto) - n + 1,
                stop = nchar(texto))
  return(out)
}

codigos_4u <- direita(texto = codigos, n = 4)
codigos_4u
## [1] "1234" "4321" "5678" "8765"


Extraindo os caracteres antes da barra de forma automatizada:

codigos2 <- c("003/2020", "1242/2020", "13579/2020")
prefixos <- substr(x = codigos2, 
                   start = 1,
                   stop = as.numeric(regexpr(pattern = "/", text = codigos2)) - 1)
prefixos
## [1] "003"   "1242"  "13579"


Separando o texto em função de um caracter:

data_texto <- c("20/04/2020")
data_texto
## [1] "20/04/2020"
data_sep <- strsplit(x = data_texto, split = "/")
data_sep
## [[1]]
## [1] "20"   "04"   "2020"
class(data_sep)
## [1] "list"
